WebSocket API

消息格式

{
  "type": "prediction" | "progress" | "error" | "notification",
  "payload": { ... },
  "timestamp": "2026-05-19T10:00:00Z",
  "requestId": "pred_xxx"
}

订阅频道

predictions
预测相关消息
rules
规则相关消息
system
系统通知

响应格式

统一响应结构

{
  "success": true,
  "data": { ... },
  "error": null,
  "message": "操作成功",
  "timestamp": "2026-05-19T10:00:00Z",
  "requestId": "req_xxx"
}

代码示例

JavaScript/TypeScript

import { PredictionService, RuleService } from '@tianwen/api';

const predictionService = new PredictionService();

// 创建预测
const result = await predictionService.predict({
  question: '今日财运如何？',
  category: 'wealth',
  system: 'meihua',
  mode: 'single'
});

if (result.success) {
  console.log('预测结果:', result.data.summary);
}

// 获取规则
const ruleService = new RuleService();
const rules = await ruleService.getRulesByCategory('bazi');
console.log('八字规则:', rules.data.length);

WebSocket 客户端

const ws = new WebSocket('ws://localhost:3000/ws');

ws.onopen = () => {
  console.log('Connected to Tianwen WebSocket');
  ws.send(JSON.stringify({
    action: 'subscribe',
    channel: 'predictions'
  }));
};

ws.onmessage = (event) => {
  const message = JSON.parse(event.data);
  console.log('收到消息:', message);
  
  if (message.type === 'prediction') {
    console.log('预测结果:', message.payload);
  } else if (message.type === 'progress') {
    console.log('进度:', message.payload);
  }
};
